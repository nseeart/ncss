loading

配置在 config/_loading.scss

$loading-width:3rem; //外盒子大小
$loading-radius:.25rem; //外盒子圆角
$loading-bgcolor:rgba(0,0,0,.5); ////外盒子背景
$loading-color:#fff; //loading颜色

    top: 50%;
    left: 50%;
    transform: translate(-50%,-50%);
}
